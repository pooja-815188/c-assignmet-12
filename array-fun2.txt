#include<stdio.h>
int fun(int row,int col,int (*ptr)[col])
{
    int i,j;
    printf("array elemets are\n");
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            printf("%d ",*(*(ptr+i)+j));
        }
        printf("\n");
    }
}
int main()
{
    int row,col;
    printf("enter a row and col\n");
    scanf("%d %d",&row,&col);
    int arr[row][col];
    int (*ptr)[col]=arr;
    int i,j;
    printf("enter a array elemets\n");
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            scanf("%d",(*(ptr+i)+j));
        }
    }
    fun(row,col,arr);
}
